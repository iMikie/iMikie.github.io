In the following sections let: s = "hello world"

indexing -- yes, just like arrays

s[i] #=> ith char s[-2] #=>"l" 2nd to last s[2..6] #=>"llo w" s[2,3] #=>"llo" [idx,cnt]

string operators -- like arrays

s << "woot" s + "woot" s.concat(more_word) s * 3   #=> s + s + s
s =~ /world/   #=> 6, idx of match s["world"]   #=> "world", rtn search string if found
==, >, <, >=, <=   #compare ascii, rtn true/false s <=> s2   #compare ascii, return -1 0 1
num = 2; "Love me #{num} times"     #=> "Love me 2 times", Object must have a .to_s method

methods

s.chars(str)   #=> array of chars s.chomp   #=> e.g. gets.chomp, remove EOL chars
s.count(str)        #sum each time any ch in str appears s.delete(str) #rtn string w/out all occ. of str
s. delete!(str) #same but modify s
[].empty?     #=> true if string is empty [].end_with?(str)     #=> true if string ends with str
[].include?(str)     #=> true if string contains str s.index(str)   #=> index of 1st occurrence
s.insert(3, "aaa") #=> "hellaaao world" (s is modified) s.length
s.size    #=> length of string
s.lines #=> break at \n, rtn array
s. lines(". ") #=>break on ". " instead 		'c'.ord, #=> ascii number for char
s.reverse, s.reverse! #=> s reversed s.scan(/w+/) #=>break on regex instead
s.split #=> break at ' ', rtn array
s.split(";\n") #=>break on ";\n" instead 		[].start_with?(str)     #=> true if string starts with str
s.strip, s.strip! #=> "remove leading whitespace s.to_i #=> convert to Integer
s.to_f #=> convert to float s.upcase, s.upcase! #=> all chars upcased.

iterators

s.each_char{|ch| puts ch}   #do block to each char s.each_line(delim){|line| puts line}   #=> if delim missing, default is crlf
s.gsub (search, replace)   #=> replace all occ. of search with replace
s.gsub!(search, replace)   #same but modifies self
